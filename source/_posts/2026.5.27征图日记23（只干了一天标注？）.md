## 最长有效括号（栈解法）
```c++
class Solution {
public:
    int longestValidParentheses(string s) {
        int ans = 0;
        stack<int> stk;
        stk.push(-1);

        for (int i = 0; i < s.size(); ++i) {
            // 左括号直接入栈
            if (s[i] == '(') {
                stk.push(i);
            }
            // 当前是右括号
            else {
                stk.pop();
                if (stk.empty()) {
                    stk.push(i);
                } else {
                    ans = max(ans, i - stk.top());
                }
            }
        }
        return ans;
    }
};
```
